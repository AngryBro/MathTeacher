Равносильные переходы

  • \( \LARGE = \)
  • \( \LARGE \neq \)
  • \( \LARGE > \normalsize,~ \LARGE< \normalsize,~ \LARGE \geq \normalsize,~ \LARGE \leq \)
  • \( \LARGE \{ \normalsize,~ \LARGE [ \)

\( \Huge = \)

  1. \( \displaystyle a_n \cdot x^n + a_{n-1} \cdot x^{n-1} + \ldots + a_2 \cdot x^2 + a_1 \cdot x + a_0 = 0 \Leftrightarrow \left[ \begin{array}{l} x=x_1 \\ x=x_2 \\ \ldots \\ x=x_n \end{array} \right.\)
  2. \( \displaystyle a_1 \cdot a_2 \cdot \ldots \cdot a_n = 0 \Leftrightarrow \left[ \begin{array}{l} \left\{ \begin{array}{l} a_1=0 \\ D(a_2),~D(a_3),~\ldots,~D(a_n) \end{array} \right. \\ \left\{ \begin{array}{l} a_2=0 \\ D(a_1),~D(a_3),~\ldots,~D(a_n) \end{array} \right.\\ \ldots \\ \left\{ \begin{array}{l} a_n=0 \\ D(a_1),~D(a_2),~\ldots,~D(a_{n-1}) \end{array} \right. \end{array} \right. \Leftrightarrow \left\{ \begin{array}{l} \left[ \begin{array}{l} a_1 = 0 \\ a_2 = 0 \\ \ldots \\ a_n = 0 \end{array} \right. \\ D(a_1),~D(a_2),~ \ldots,~ D(a_n) \end{array} \right. \)
  3. \( \displaystyle \frac{a}{b} = 0 \Leftrightarrow \left\{ \begin{array}{l} a = 0 \\ b \neq 0 \end{array} \right. \)
  4. \( \displaystyle \operatorname{monotone}{(a)} = \operatorname{monotone}{(b)} \Leftrightarrow \left\{ \begin{array}{l} a=b \\ a \in D(\operatorname{monotone}) \end{array} \right. \)
  5. \( \displaystyle \sqrt[2n]{a} = b \Leftrightarrow \left\{ \begin{array}{l} a = b^{2n} \\ b \geq 0 \end{array} \right. \)
  6. \( \displaystyle a_n \cdot \sin^n{x} + a_{n-1} \cdot \sin^{n-1}{x} \cdot \cos{x} + a_{n-2} \cdot \sin^{n-2} \cdot \cos^2{x} + \ldots + a_1 \cdot \sin{x} \cdot \cos^{n-1} + a_0 \cdot \cos^n{x} = 0 \Leftrightarrow \)

    \( \Leftrightarrow a_n \cdot \operatorname{tg}^{n}{x} + a_{n-1} \cdot \operatorname{tg}^{n-1}{x} + \ldots + a_1 \cdot \operatorname{tg}{x} + a_0 = 0 \)

\( \Huge \neq \)

  1. \( a_1 \cdot a_2 \cdot \ldots \cdot a_n \neq 0 \Leftrightarrow \left\{ \begin{array}{l} a_1 \neq 0 \\ a_2 \neq 0 \\ \ldots \\ a_n \neq 0 \end{array} \right. \)
  2. \( \displaystyle \frac{a}{b} \neq 0 \Leftrightarrow \left\{ \begin{array}{l} a \neq 0 \\ b \neq 0 \end{array} \right. \)
  3. \( \operatorname{monotone}{(a)} \neq \operatorname{monotone}{(b)} \Leftrightarrow \left\{ \begin{array}{l} a \neq b \\ a \in D(\operatorname{monotone}) \\ b \in D(\operatorname{monotone}) \end{array} \right. \)

\( \Huge > \normalsize,~ \Huge< \normalsize,~ \Huge \geq \normalsize,~ \Huge \leq \)

  1. \( \displaystyle \frac{(x-a_1)^{k_1} \cdot (x-a_2)^{k_2} \cdot \ldots \cdot (x-a_n)^{k_n}} {(x-b_1)^{t_1} \cdot (x-b_2)^{t_2} \cdot \ldots \cdot (x-b_m)^{t_m}} \vee 0 \stackrel{м.~и.}{\Longleftrightarrow} \)
  2. \( \operatorname{increasing}{(a)} \vee \operatorname{increasing}{(b)} \Leftrightarrow \left\{ \begin{array}{l} a \vee b \\ a \in D(\operatorname{increasing}) \\ b \in D(\operatorname{increasing}) \end{array} \right. \)
  3. \( \operatorname{decreasing}{(a)} \vee \operatorname{decreasing}{(b)} \Leftrightarrow \left\{ \begin{array}{l} a \land b \\ a \in D(\operatorname{decreasing}) \\ b \in D(\operatorname{decreasing}) \end{array} \right. \)
  4. \( \displaystyle c \cdot ( \operatorname{increasing}{(a)} - \operatorname{increasing}{(b)} )^{\pm n} \vee 0 \Leftrightarrow \left\{ \begin{array}{l} c \cdot (a - b)^{\pm n} \vee 0 \\ a \in D(\operatorname{increasing}) \\ b \in D(\operatorname{increasing}) \end{array} \right. \)

\( \Huge \{ \normalsize,~ \Huge [ \)

  1. \( \displaystyle false \Leftrightarrow x \not \in \mathbb{R} \)
  2. \( true \Leftrightarrow D(true) \)
  3. \( \left\{ \begin{array}{l} A \\ A \end{array} \right. \Leftrightarrow A \)
  4. \( \left[ \begin{array}{l} A \\ A \end{array} \right. \Leftrightarrow A \)
  5. \( \left\{ \begin{array}{l} A \\ x \not \in \mathbb{R} \end{array} \right. \Leftrightarrow x \not \in \mathbb{R} \)
  6. \( \left\{ \begin{array}{l} A \\ x \in \mathbb{R} \end{array} \right. \Leftrightarrow A \)
  7. \( \left[ \begin{array}{l} A \\ x \not \in \mathbb{R} \end{array} \right. \Leftrightarrow A \)
  8. \( \left[ \begin{array}{l} A \\ x \in \mathbb{R} \end{array} \right. \Leftrightarrow x \in \mathbb{R} \)
  9. \( \left\{ \begin{array}{l} a = b \\ F(a) \end{array} \right. \Leftrightarrow \left\{ \begin{array}{l} a = b \\ F(b) \end{array} \right. \)
  10. \( \left\{ \begin{array}{l} A \Leftrightarrow B \\ A \end{array} \right. \Leftrightarrow \left\{ \begin{array}{l} A \Leftrightarrow B \\ B \end{array} \right. \)
  11. \( A \Leftrightarrow \left[ \begin{array}{l} \left\{ \begin{array}{l} CASE_1 \\ A \end{array} \right. \\ \left\{ \begin{array}{l} CASE_2 \\ A \end{array} \right. \\ \ldots \\ \left\{ \begin{array}{l} CASE_n \\ A \end{array} \right. \\ \end{array} \right. \)
  12. \( \left\{ \begin{array}{l} \left[ \begin{array}{l} A \\ B \end{array} \right. \\ \left[ \begin{array}{l} A \\ C \end{array} \right. \\ \end{array} \right. \Leftrightarrow \left[ \begin{array}{l} A \\ \left\{ \begin{array}{l} B \\ C \end{array} \right. \end{array} \right. \)
  13. \( \left[ \begin{array}{l} \left\{ \begin{array}{l} A \\ B \end{array} \right. \\ \left\{ \begin{array}{l} A \\ C \end{array} \right. \\ \end{array} \right. \Leftrightarrow \left\{ \begin{array}{l} A \\ \left[ \begin{array}{l} B \\ C \end{array} \right. \end{array} \right. \)
