\( \Huge = \)

  1. \( \begin{array}[t]{|l|} \hline \displaystyle a_n \cdot x^n + a_{n-1} \cdot x^{n-1} + \ldots + a_2 \cdot x^2 + a_1 \cdot x + a_0 = 0 \Leftrightarrow \\ \Leftrightarrow a_n \cdot (x - x_1) \cdot(x - x_2) \cdot \ldots \cdot (x - x_m) = 0 \Leftrightarrow \left[ \begin{array}{l} x=x_1 \\ x=x_2 \\ \ldots \\ x=x_m \end{array} \right. \\ \hline \end{array} \)
    • \( a \cdot x^2 + b \cdot x + c = 0 \Leftrightarrow \left[ \begin{array}{l} x = x_1 \\ x = x_2 \end{array} \right. \)
  2. \( \begin{array}[t]{|l|} \hline \displaystyle a_1 \cdot a_2 \cdot \ldots \cdot a_n = 0 \Leftrightarrow \left[ \begin{array}{l} \left\{ \begin{array}{l} a_1=0 \\ ограничения~из~a_2,~a_3, \ldots ,a_n \end{array} \right. \\ \left\{ \begin{array}{l} a_2=0 \\ ограничения~из~a_1,~a_3, \ldots ,a_n \end{array} \right.\\ \ldots \\ \left\{ \begin{array}{l} a_n=0 \\ ограничения~из~a_1,~a_2, \ldots ,a_{n-1} \end{array} \right. \end{array} \right. \Leftrightarrow \left\{ \begin{array}{l} \left[ \begin{array}{l} a_1 = 0 \\ a_2 = 0 \\ \ldots \\ a_n = 0 \end{array} \right. \\ ограничения~из~a_1,~a_2, \ldots ,a_n \end{array} \right. \\ \hline \end{array} \)
    • \( a \cdot b = 0 \Leftrightarrow \left[ \begin{array}{l} \left\{ \begin{array}{l} a = 0 \\ ограничения~из~b \end{array} \right. \\ \left\{ \begin{array}{l} b = 0 \\ ограничения~из~a \end{array} \right. \end{array} \right. \Leftrightarrow \left\{ \begin{array}{l} \left[ \begin{array}{l} a = 0 \\ b = 0 \end{array} \right.\\ ограничения~из~a,~b \end{array} \right. \)
  3. \(\begin{array}[t]{|l|} \hline \displaystyle \frac{a}{b} = \frac{c}{d} \Leftrightarrow \left\{ \begin{array}{l} a \cdot d = b \cdot c \\ b \neq 0 \\ d \neq 0 \end{array} \right. \\ \hline \end{array} \)
    • \( \displaystyle \frac{a}{b} = 0 \Leftrightarrow \left\{ \begin{array}{l} a = 0 \\ b \neq 0 \end{array} \right. \)
  4. \(\begin{array}[t]{|l|} \hline \displaystyle \operatorname{МонотоннаяФункция}{(a)} = \operatorname{МонотоннаяФункция}{(b)} \Leftrightarrow \left\{ \begin{array}{l} a=b \\ ограничение~на~a~или~b \end{array} \right. \\ \hline \end{array} \)
    • \( (\not 1)^a = (\not 1)^b \Leftrightarrow a = b \)
    • \( \log{a} = \log{b} \Leftrightarrow \left\{ \begin{array}{l} a = b \\ a > 0 \end{array} \right. \)
    • \( \sqrt[чёт]{a} = \sqrt[чёт]{b} \Leftrightarrow \left\{ \begin{array}{l} a = b \\ a \geq 0 \end{array} \right. \)
    • \( \sqrt[нечёт]{a} = \sqrt[нечёт]{b} \Leftrightarrow a = b \)
    • \( a^{нечёт} = b^{нечёт} \Leftrightarrow a = b \)
  5. \(\begin{array}[t]{|l|} \hline \displaystyle \sqrt[чёт]{a} = b \Leftrightarrow \left\{ \begin{array}{l} a = b^{чёт} \\ b \geq 0 \end{array} \right. \\ \hline \end{array} \)
    • \( \sqrt{a} = b \Leftrightarrow \left\{ \begin{array}{l} a = b^2 \\ b \geq 0 \end{array} \right. \)
  6. \( \begin{array}[t]{|l|} \hline \displaystyle a_n \cdot \sin^n{x} + a_{n-1} \cdot \sin^{n-1}{x} \cdot \cos{x} + a_{n-2} \cdot \sin^{n-2} \cdot \cos^2{x} + \ldots + a_1 \cdot \sin{x} \cdot \cos^{n-1} + a_0 \cdot \cos^n{x} = 0 \Leftrightarrow \\ \Leftrightarrow a_n \cdot \operatorname{tg}^{n}{x} + a_{n-1} \cdot \operatorname{tg}^{n-1}{x} + \ldots + a_1 \cdot \operatorname{tg}{x} + a_0 = 0 \\ \hline \end{array} \)
    • \( a \cdot \sin{x} + b \cdot \cos{x} = 0 \Leftrightarrow a \cdot \operatorname{tg}{x} + b = 0 \)
    • \( a \cdot \sin^2{x} + b \cdot \sin{x} \cdot \cos{x} + c \cdot \cos^2{x} = 0 \Leftrightarrow a \cdot \operatorname{tg^2}{x} + b \cdot \operatorname{tg}{x} + c = 0 \)

\( \Huge \neq \)

  1. \( \begin{array}[t]{|l|} \hline a_1 \cdot a_2 \cdot \ldots \cdot a_n \neq 0 \Leftrightarrow \left\{ \begin{array}{l} a_1 \neq 0 \\ a_2 \neq 0 \\ \ldots \\ a_n \neq 0 \end{array} \right. \\ \hline \end{array} \)
  2. \( \begin{array}[t]{|l|} \hline \displaystyle \frac{a}{b} \neq \frac{c}{d} \Leftrightarrow \left\{ \begin{array}{l} a \cdot d \neq b \cdot c \\ b \neq 0 \\ d \neq 0 \end{array} \right. \\ \hline \end{array} \)
    • \( \displaystyle \frac{a}{b} \neq 0 \Leftrightarrow \left\{ \begin{array}{l} a \neq 0 \\ b \neq 0 \end{array} \right. \)
  3. \( \begin{array}[t]{|l|} \hline \operatorname{МонотоннаяФункция}{(a)} \neq \operatorname{МонотоннаяФункция}{(b)} \Leftrightarrow \left\{ \begin{array}{l} a \neq b \\ ограничение~на~a \\ ограничение~на~b \end{array} \right. \\ \hline \end{array} \)
    • \( (\not 1)^a \neq (\not 1)^b \Leftrightarrow a \neq b \)
    • \( \log{a} \neq \log{b} \Leftrightarrow \left\{ \begin{array}{l} a \neq b \\ a > 0 \\ b > 0 \end{array} \right. \)
    • \( \sqrt[чёт]{a} \neq \sqrt[чёт]{b} \Leftrightarrow \left\{ \begin{array}{l} a \neq b \\ a \geq 0 \\ b \geq 0 \end{array} \right. \)
    • \( \sqrt[нечёт]{a} \neq \sqrt[нечёт]{b} \Leftrightarrow a \neq b \)
    • \( a^{нечёт} \neq b^{нечёт} \Leftrightarrow a \neq b \)
